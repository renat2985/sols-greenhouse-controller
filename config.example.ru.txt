# Sols Greenhouse Controller config.txt
# Файл должен лежать в корне MicroSD карты: /config.txt
# Строки с # являются комментариями.

# Название Wi-Fi роутера и пароль.
ssid=YourRouterName
password=YourRouterPassword

# Яркость экрана от 0 до 255.
brightness=180

# Имя устройства в cloud-панели.
device=Greenhouse

# Отправка данных в облако sols.lv.
cloud_enabled=true
cloud_url=http://sols.lv

# Публичная ссылка на панель клиента.
# Если оставить пустым, прошивка создаст длинный токен из ESP32 ID.
cloud_public_token=

# Приватный API-токен для отправки данных.
# Не публикуйте его на GitHub и не отправляйте чужим людям.
# Если оставить пустым, прошивка создаст длинный токен из ESP32 ID.
cloud_api_token=

# Как часто отправлять данные на сервер, в секундах.
# Минимальный безопасный лимит: 30 секунд.
cloud_interval_seconds=300

# Целевая температура в теплице.
target_temp_c=26

# Гистерезис, чтобы вентиляторы не включались слишком часто.
temp_hysteresis_c=1

# Максимальная влажность в процентах.
# Если внутри слишком влажно, а снаружи суше, включится вытяжка.
humidity_max=80

# Минимальная влажность в процентах. Зарезервировано для будущей логики.
humidity_min=45

# Какую зону контролирует автоматика.
# inside = датчик на экране/плате контроллера.
# outside = выносной датчик на проводе, например погреб или подвал.
control_zone=inside

# Датчик влажности почвы.
# false = датчик не подключен, полив идет по расписанию.
# true = полив идет по влажности почвы.
soil_sensor_enabled=false

# Инвертировать проценты датчика почвы, если датчик работает наоборот.
soil_sensor_invert=false

# Калибровка сухого/мокрого значения датчика почвы, ADC.
soil_dry_raw=3200
soil_wet_raw=1300

# При какой влажности почвы включать полив.
soil_moisture_min=45

# Желаемая влажность почвы. Показывается пользователю.
soil_moisture_target=60

# Сколько секунд держать реле воды включенным.
watering_seconds=15

# Пауза между поливами, если используется датчик почвы.
soil_watering_cooldown_min=120

# Полив по расписанию, если датчик почвы не подключен.
watering_schedule_enabled=true

# Интервал полива в минутах.
# 720 = каждые 12 часов.
watering_interval_min=720

# Большинство модулей реле работают как active-low.
relay_active_low=true

# Режим второго реле.
# intake = приточный вентилятор.
# light = подсветка для рассады/растений по датчику света.
relay2_mode=intake

# Настройки подсветки, если relay2_mode=light.
# Часы указываются по UTC времени.
grow_light_schedule_enabled=true
grow_light_start_hour=6
grow_light_end_hour=22
grow_light_min_percent=55
grow_light_hysteresis_percent=8

# Инвертировать проценты датчика света, если нужно.
light_invert=false
